def check_missing(dataset, name="Dataset"):
    print(f"\nMissing Values in {name}:")
    
    features_with_na = [feature for feature in dataset.columns if dataset[feature].isnull().sum() > 0]
    
    for feature in features_with_na:
        percent_missing = np.round(dataset[feature].isnull().mean() * 100, 2)
        print(f"{feature}: {percent_missing}% missing values")
    
    if len(features_with_na) == 0:
        print("✅ No missing values found.")

# Check in training and testing feature sets
check_missing(X_train, "X_train (features)")
check_missing(X_test, "X_test (features)")

# For y (target) we check separately
print("\nMissing in y_train:", y_train.isnull().sum())
print("Missing in y_test:", y_test.isnull().sum())

import pandas as pd
from sklearn.linear_model import LinearRegression

# 1️ Assuming you have your dataset loaded as df
# Features and target
X = df[['Square_Footage', 'Num_Bedrooms', 'Num_Bathrooms', 'Year_Built', 
        'Lot_Size', 'Garage_Size', 'Neighborhood_Quality']]
y = df['House_Price']

# 2️ Train Linear Regression
lin_model = LinearRegression()
lin_model.fit(X, y)

# 3️ Define prediction function
def predict_house_price(square_ft, bedrooms, bathrooms, year_built, lot_size, garage_size, neighborhood_quality):
    features = pd.DataFrame([[square_ft, bedrooms, bathrooms, year_built, lot_size, garage_size, neighborhood_quality]],
                            columns=['Square_Footage', 'Num_Bedrooms', 'Num_Bathrooms', 'Year_Built',
                                     'Lot_Size', 'Garage_Size', 'Neighborhood_Quality'])
    predicted_price = lin_model.predict(features)[0]
    return round(predicted_price, 2)

# 4️ Example usage
predicted = predict_house_price(
    square_ft=2500, 
    bedrooms=3, 
    bathrooms=2, 
    year_built=2010, 
    lot_size=5000, 
    garage_size=2, 
    neighborhood_quality=7
)

print("Predicted House Price: $", predicted)
